using System.ComponentModel.DataAnnotations;

namespace WebAppFullname.Models
{
    public class ModelFn
    {
        public string? Firstname { get; set; }
        public string? Lastname {get; set; }
    }
}

right click on controller add controller ->controller empty->name:UserController->add

UserController.cs

using Microsoft.AspNetCore.Mvc;

namespace WebAppFullname.Controllers
{
    public class UserController : Controller
    {
        public IActionResult Index()
        {
            return View();
        }
        [HttpPost]
        public IActionResult Index(Models.ModelFn model)
        {
            if(ModelState.IsValid)
            {
                ViewBag.Message=model.Firstname + model.Lastname;
                return View();

            }
            return View();
        }
    }
}
