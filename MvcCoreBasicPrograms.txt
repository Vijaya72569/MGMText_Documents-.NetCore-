UserModel.cs

using System.ComponentModel.DataAnnotations;

namespace WebAppFname.Models
{
    public class UserModel
    {
      

        [Required]
        [Display(Name = "First Name")]
        public string? FirstName { get; set; }

        [Required]
        [Display(Name = "Last Name")]
        public string? LastName { get; set; }

        //public string FullName => $"{FirstName} {LastName}";
        public string FullName => FirstName + LastName;
    }

}

using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Server.HttpSys;
using System;
using WebAppFname.Models;

namespace WebAppFname.Controllers
{
    public class UserController : Controller
    {
        public IActionResult Index()
        {
            return View();
        }
        [HttpPost]
        public IActionResult Index(UserModel model)
        {
            if (ModelState.IsValid)
            {
                ViewBag.FullName = model.FullName;
            }
            return View(model);
        }
           
       
    }
}

Maths.cs

namespace WebAppFname.Models
{
    public class Maths
    {
        public int num1 {  get; set; }
        public int num2 { get; set; }
        public  int sum=>num1 + num2;
        public int sub=>num1 - num2;
        public int mul=>num1 * num2;
        public int div=>num1 / num2;

    }
}


MathsController.cs

using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Server.HttpSys;
using System;
using WebAppFname.Models;

namespace WebAppFname.Controllers
{
    public class MathsController : Controller
    {

  public IActionResult Index() 
        { 
            return View();
        }
        [HttpPost]
        public IActionResult Index(Maths model ,string str)
        {
            if (ModelState.IsValid)
            {
                if (str == "Add")
                {
                    ViewBag.sum = model.sum;
                }
               else if (str == "Sub")
                {
                    ViewBag.sub = model.sub;
                }
                else if (str == "Mul")
                {
                    ViewBag.mul = model.mul;
                }
                else if (str == "Div")
                {
                    ViewBag.div = model.div;
                }
               
            }
            return View(model);
            
        }
                  
    }
}
